\newcommand{\weekpageMonToWed}{%
  \setcounter{page}{\monPage}%

  \begin{tikzpicture}[
      gray,
      ultra thin,
      field/.style={
        anchor=south west,
        minimum height=\rowheight,
        rectangle,
      },
      time/.style={field, minimum width=17.5mm, black},
      report/.style={field, minimum width=11mm, black!10},
      title/.style={
        anchor=north west,
        minimum width=\fromright{0},
        text width=\fromright{1em},
        rectangle split,
        rectangle split parts=2,
        black,
      },
    ]

  \node at (0,\fromtop{0}) [title] {%
    \footnotesize%
    \textcolor{lgray}{\monDay}%
      \hfill%
    \textcolor{lgray}{W\monWeek}
      \hfill%
      \textcolor{lgray}{\monDate}\,/\,{\monDayOfWeek}};

  \foreach \T [count=\r] in {7,...,0} {
    \draw [dashed] (0,\r*2\rowheight) -- %
        (\fromright{0},\r*2\rowheight);
    \draw [dotted, thick] (0,\r*2\rowheight+\rowheight) -- %
        (\fromright{0},\r*2\rowheight+\rowheight);
    \node at (0,\r*2\rowheight+\rowheight) [time] {};
  }
  \def\distA{18\rowheight}%

  \node at (0,\fromtop{19\rowheight}) [title] {%
    \footnotesize%
    \textcolor{lgray}{\tueDay}%
      \hfill%
    \textcolor{lgray}{W\tueWeek}
      \hfill%
      \textcolor{lgray}{\tueDate}\,/\,{\tueDayOfWeek}};

  \foreach \T [count=\r] in {7,...,0} {
    \draw [dashed] (0,\distA+\r*2\rowheight) -- %
        (\fromright{0},\distA+\r*2\rowheight);
    \draw [dotted, thick] (%
      0,\distA+\r*2\rowheight+\rowheight) -- %
        (\fromright{0},\distA+\r*2\rowheight+\rowheight);
    \node at (0,\distA+\r*2\rowheight+\rowheight) [time] {};
  }
  \def\distB{36\rowheight}%

  \node at (0,\fromtop{37\rowheight}) [title] {%
    \footnotesize%
    \textcolor{lgray}{\wedDay}%
      \hfill%
    \textcolor{lgray}{W\wedWeek}
      \hfill%
      \textcolor{lgray}{\wedDate}\,/\,{\wedDayOfWeek}};

  \foreach \T [count=\r] in {8,...,0} {
    \draw [dashed] (0,\distB+\r*2\rowheight) -- %
        (\fromright{0},\distB+\r*2\rowheight);
    \draw [dotted, thick] (%
      0,\distB+\r*2\rowheight+\rowheight) -- %
        (\fromright{0},\distB+\r*2\rowheight+\rowheight);
    \node at (0,\distB+\r*2\rowheight+\rowheight) [time] {};
  }

  \draw [dashed] (0,0) rectangle (%
    \fromright{0},57\rowheight);
  \end{tikzpicture}
}
